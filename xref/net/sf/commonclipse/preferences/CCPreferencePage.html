18  
19  import net.sf.commonclipse.CCMessages;
20  import net.sf.commonclipse.CCPlugin;
21  
22  import org.eclipse.jface.preference.BooleanFieldEditor;
23  import org.eclipse.jface.preference.RadioGroupFieldEditor;
24  import org.eclipse.ui.IWorkbench;
25  import org.eclipse.ui.IWorkbenchPreferencePage;

28  /**
29   * This class represents a preference page that is contributed to the Preferences dialog. By subclassing <code>
30   * FieldEditorPreferencePage</code>,
31   * we can use the field support built into JFace that allows us to create a page that is small and knows how to save,
32   * restore and apply itself.
33   * <p>
34   * This page is used to modify preferences only. They are stored in the preference store that belongs to the main
35   * plug-in class. That way, preferences can be accessed directly via the preference store.
36   * @author fgiust
37   * @version $Revision$ ($Author$)
38   */
39  
40  public class CCPreferencePage extends TabbedFieldEditorPreferencePage implements IWorkbenchPreferencePage
41  {
42  
43      /**
44       * init preference page.
45       */
46      public CCPreferencePage()
47      {
48          super(GRID);
49          setPreferenceStore(CCPlugin.getDefault().getPreferenceStore());
50      }
51  
52      /**
53       * Creates the field editors. Field editors are abstractions of the common GUI blocks needed to manipulate various
54       * types of preferences. Each field editor knows how to save and restore itself.
55       */
56      @Override
57      public void createFieldEditors()
58      {
59  
60          addTab(CCMessages.getString("preference.tab.tostring")); //$NON-NLS-1$
61  
62          addField(new RadioGroupFieldEditor(
63              CCPlugin.P_TOSTRING_BEAN,
64              CCMessages.getString("preference.tostringtype"), 1, new String[][]{ //$NON-NLS-1$
65              {CCMessages.getString("preference.tostringtype.bean"), CCPlugin.TOSTRINGSTYLE_BEAN}, //$NON-NLS-1$
66                  {CCMessages.getString("preference.tostringtype.fieds"), //$NON-NLS-1$
67                      CCPlugin.TOSTRINGSTYLE_FIELDS}},
68              getFieldEditorParent(),
69              true));
70  
71          addField(new SpacerFieldEditor(getFieldEditorParent()));
72  
73          ClassConstantFieldEditor selectEditor = new ClassConstantFieldEditor(CCPlugin.P_TOSTRING_STYLE, CCMessages
74              .getString("preference.customtostringtype"), //$NON-NLS-1$
75              getFieldEditorParent());
76  
77          String[] styles = new String[]{"org.apache.commons.lang.builder.ToStringStyle.DEFAULT_STYLE", //$NON-NLS-1$
78              "org.apache.commons.lang.builder.ToStringStyle.MULTI_LINE_STYLE", //$NON-NLS-1$
79              "org.apache.commons.lang.builder.ToStringStyle.NO_FIELD_NAMES_STYLE", //$NON-NLS-1$
80              "org.apache.commons.lang.builder.ToStringStyle.SIMPLE_STYLE"}; //$NON-NLS-1$
81  
82          selectEditor.setPredefinedValues(styles);
83  
84          selectEditor.setErrorMessage(CCMessages.getString("preference.customtostring.error")); //$NON-NLS-1$
85  
86          addField(selectEditor);
87  
88          addField(new SpacerFieldEditor(getFieldEditorParent()));
89  
90          addTab(CCMessages.getString("preference.tab.general")); //$NON-NLS-1$
91  
92          addField(new BooleanFieldEditor(CCPlugin.P_TOSTRING_SUPER, CCMessages
93              .getString("preference.appendsuper.tostring"), getFieldEditorParent())); //$NON-NLS-1$
94          addField(new BooleanFieldEditor(CCPlugin.P_HASHCODE_SUPER, CCMessages
95              .getString("preference.appendsuper.hashcode"), getFieldEditorParent())); //$NON-NLS-1$
96          addField(new BooleanFieldEditor(
97              CCPlugin.P_EQUALS_SUPER,
98              CCMessages.getString("preference.appendsuper.equals"), getFieldEditorParent())); //$NON-NLS-1$
99          addField(new BooleanFieldEditor(CCPlugin.P_COMPARETO_SUPER, CCMessages
100             .getString("preference.appendsuper.compareto"), //$NON-NLS-1$
101             getFieldEditorParent()));
102 
103         addField(new SpacerFieldEditor(getFieldEditorParent()));
104 
105         addField(new BooleanFieldEditor(CCPlugin.P_FINALPARAMETERS, //
106             CCMessages.getString("preference.finalparameters"), //$NON-NLS-1$
107             getFieldEditorParent()));
108 
109         addField(new BooleanFieldEditor(CCPlugin.P_EQUALS_INSTANCECHECK, CCMessages
110             .getString("preference.equals.equalitycheck"), //$NON-NLS-1$
111             getFieldEditorParent()));
112 
113         addField(new BooleanFieldEditor(CCPlugin.P_OVERRIDE_ANNOTATION, CCMessages
114                 .getString("preference.overrideannotation"), //$NON-NLS-1$
115                 getFieldEditorParent()));
116         
117         addField(new SpacerFieldEditor(getFieldEditorParent()));
118 
119         addField(new BooleanFieldEditor(CCPlugin.P_DONTASKONOVERWRITE, CCMessages
120             .getString("preference.overwriteconfirmation"), //$NON-NLS-1$
121             getFieldEditorParent()));
122         
123         addField(new SpacerFieldEditor(getFieldEditorParent()));
124 
125         addTab(CCMessages.getString("preference.tab.exclude")); //$NON-NLS-1$
126 
127         AddRemoveListFieldEditor excludedEditor = new AddRemoveListFieldEditor(CCPlugin.P_EXCLUDE, CCMessages
128             .getString("preference.excluded"), //$NON-NLS-1$
129             getFieldEditorParent());
130 
131         addField(excludedEditor);
132 
133     }

140     {
141     }
